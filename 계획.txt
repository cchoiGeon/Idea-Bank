테이블 :
1. 회원가입 
CREATE TABLE register(
num INT(11) NOT NULL AUTO_INCREMENT,
id TEXT NOT NULL,
password TEXT NOT NULL,
name TEXT NOT NULL,
phone TEXT NULL,
PRIMARY KEY(num)
);
CREATE TABLE idea_board(
num INT(11) NOT NULL AUTO_INCREMENT,
writer TEXT NOT NULL,
title TEXT NOT NULL,
content TEXT NOT NULL,
time DATETIME NULL,
PRIMARY KEY(num)
);
////
CREATE TABLE idea_comments(
num INT(11) NOT NULL AUTO_INCREMENT
board_num INT(11) NOT NULL,
writer TEXT NOT NULL,
title TEXT NOT NULL,
comments TEXT NULL,
PRIMARY KEY(num)
);
